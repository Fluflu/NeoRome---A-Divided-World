namespace = rome

rome.1 = {
	type = country_event
	placement = ROOT
	
	title = rome.1.t
	desc = rome.1.d
	flavor = rome.1.f
	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	immediate = {
		if = {
			limit = {
				c:ALB = { is_subject_of = c:PAP }
			}
 			c:PAP = {
 				annex = c:ALB
			}	
		}
		if = {
			limit = {
				c:SIC = { is_subject_of = c:PAP }
			}
 			c:PAP = {
 				annex = c:SIC
			}	
		}
    	}

   	 trigger = {
     	   	exists = c:PAP
   	     	this = c:PAP
		year > 1840.1.1
		NOT = { exists = c:ITA }	
	}

	event_image = {
		video = "gfx/event_pictures/middleeast_courtroom_upheaval.bk2"
	}

	duration = 3
	
	option = {
		name = rome.1.a

		create_diplomatic_play = {
			name = "Epirus wars"
			war = yes
			type = dp_annex_war
			target_state = s:STATE_ALBANIA.region_state:EPR
		}
	}
}
